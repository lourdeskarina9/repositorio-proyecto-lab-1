const int pinPotenciometro = A0;
const int pinLed = 9;

void setup() {
  pinMode(pinLed, OUTPUT);
  Serial.begin(9600); // Inicia la comunicación serial
}

void loop() {
  int valorPot = analogRead(pinPotenciometro); // Lee el valor entre 0 y 1023
  int brillo = map(valorPot, 0, 1023, 0, 255); // Convierte a valor PWM
  analogWrite(pinLed, brillo); // Controla el brillo

  // Imprime valores al monitor serial
  Serial.print("Potenciometro: ");
  Serial.print(valorPot);
  Serial.print(" -> Brillo PWM: ");
  Serial.println(brillo);

  delay(100); // Pequeña pausa para facilitar la lectura
}
